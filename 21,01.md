heads = coins.count(1)
tails = coins.count(0)

# Задача 12
S, P = 5, 6
found = False
for x in range(1, 1001):
    if S - x > 0:
        y = S - x
        if x * y == P:
            print(f"Задача 12: числа {x} и {y}")
            found = True
            break
if not found:
    print("Задача 12: такие числа не найдены")
